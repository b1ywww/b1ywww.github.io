$$ \begin{align} (1)&P(X\le 150 ) = \int_{100}^{150}\frac{100}{x^2}dx=\frac{1}{3}\\
&p=(P(X >150)^3 = \frac{8}{27}\\
(2)&p=C_3^1\frac{1}{3}(\frac{2}{3})^2=\frac{4}{9}\\
(3)&当x\ge 100时F(x)=\int_{-\infty}^xf(s)ds=1-\frac{100}{x}\\
&所以F(X)=\begin{cases} 1-\frac{100}{x}, x \ge 100\\
0,x<100 \end{cases} \end{align} $$

17.

$$ \begin{align} &设F(x) = kx,(0\le x \ge a)\\
&因为F(a)-F(0)=1\\
&所以k=\frac{1}{a}\\
&F(x)=\begin{cases} 0, x < 0\\
\frac{x}{a}, 0\ge x \le a\\
1,x>a \end{cases} \end{align} $$

18.

$$ \begin{align} &由均匀分布可得密度函数为f(x)=\begin{cases} \frac{1}{3}, 2\le x \le 5\\ 0, x<2或x>5\end{cases}\\
&P(X>3)=\int_3^5\frac{1}{3}dx=\frac{2}{3}\\
&p=C_3^2(\frac{2}{3})^2\frac{1}{3}+(\frac{2}{3})^2 = \frac{20}{27} \end{align} $$

&所以\sigma\le \frac{40}{1.29}=31.25 \end{align} $$

&P(Y=1)=P(X=-1)+P(X=1)=\frac{1}{6}+\frac{1}{15}=\frac{7}{30}\\
&P(Y=4)=P(X=-2)=\frac{1}{5}\\
&P(Y=9)=P(X=3)=\frac{11}{30}\\
&所以Y的分布律为\\
&\begin{array}{c|lcr} \hline Y&0&1&4&9\\
\hline P_k&\frac{1}{5}&\frac{7}{30}&\frac{1}{5}&\frac{11}{30}\\
\hline \end{array} \end{align} $$

30

$$ \begin{align} (1)&当y\le 0时,F_Y(y)=P(Y\le y) = 0\\
&当y>0时,\\
&F_Y(y)=P(x\le y)=P(e^x\le y)= P(X\le lny)=\int_{-\infty}^{lny}f_X(x)dx\\
&所以f_Y(y)=\frac{F_Y}(y){dy}=\frac{1}{y}f_X(lny)=\frac{1}{y}\frac{e^{\frac{-(lny)^2}{2}}}{\sqrt{2\pi}}\\
(2)&Y=2X^2+1\ge 1\\
&当y\le 1时F_Y(y)=P(Y\le y )=0\\
&当y>1 时F_Y(y)=P(Y\le y)=P(X^2\le \frac{y-1}{2})=P(-\sqrt{\frac{y-1}{2}}\le X\le \sqrt{\frac{y-1}{2}})=\int_{-\sqrt{\frac{y-1}{2}}}^{\sqrt{\frac{y-1}{2}}}f_X(x)dx\\
&f_Y(y)=\frac{dF_Y(y)}{dy}=\frac{1}{4}\sqrt{\frac{2}{y-1}}(f_X(-\sqrt{\frac{y-1}{2}}+ f_X\sqrt{\frac{y-1}{2}}=\frac{1}{2}\sqrt{\frac{2}{y-1}}\frac{e^{-\frac{y-1}{4}}}{\sqrt{2\pi}}\\
(3)&Y=|X|\ge 0\\
&当y\le0时 F_Y(y)=P(Y\le y)=0\\
&当y >0时 F_Y(y)=P(|X|\le y)=P(-y\le X \le y)=\int_{-y}^{y}f_X(x)dx\\
&f_Y(y)=\frac{dF_Y(y)}{dy}=f_X(y)+f_X(-y)=\frac{2e^{-\frac{y^2}{2}}}{\sqrt{2\pi}} \end{align} $$

31

$$ \begin{align} (1)&因为0<x<1 所以1<Y<e\\
&y\le 1时F_Y(y)=0\\
&1<y<e时F_Y(y)=P(X\le lny)=\int_0^{lny}dx=lny\\
&当y\ge e时 F_Y(y) =1\\
&所以分布函数为F_Y(y)=\begin{cases}0& y\le 1\\lny&1<y<e\\1& y\ge e\end{cases}\\
&密度函数为f_Y(y)=\begin{cases}0& y\le 1或y\ge e\\ \frac{1}{y}& 1<y<e\end{cases}\\
(2)&因为0<x<1 所以Z>0\\
&当z\le 0时F_Z(z)=0\\
&当z>0时，F_Z(z)=P(x\ge e^{-\frac{z}{2}})=\int_{e^{-\frac{z}{2}}}^{1}dx=1-e^{-\frac{z}{2}}\\
&分布函数为F_Z(z)=\begin{cases} 0& z\le 0\\ 1-e^{-\frac{z}{2}}&z>0\end{cases}\\
&密度函数为f_Z(z)=\begin{cases}\frac{1}{2}e^{-\frac{z}{2}}&z>0\\0& z\le 0\end{cases} \end{align} $$
